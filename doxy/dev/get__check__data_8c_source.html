121 static
122 bool
123 fetch_column_info(
124  Column_info_t *info) {
125  /*
126  * [SPI_fnumber](https://www.postgresql.org/docs/12/spi-spi-fnumber.html)
127  */
128  info->colNumber = SPI_fnumber(SPI_tuptable->tupdesc, info->name);
129  if (info->strict && !column_found(info->colNumber)) {
130  elog(ERROR, "Column '%s' not Found", info->name);
131  }
132 
133  if (column_found(info->colNumber)) {
134  /*
135  * [SPI_gettypeid](https://www.postgresql.org/docs/12/spi-spi-gettypeid.html)
136  */
137  (info->type) = SPI_gettypeid(SPI_tuptable->tupdesc, (info->colNumber));
138  PGR_DBG("%s %ld", info->name, info->type);
139  if (SPI_result == SPI_ERROR_NOATTRIBUTE) {
140  elog(ERROR, "Type of column '%s' not Found", info->name);
141  }
142  return true;
143  }
144  return false;
145 }

251  char value = default_value;
252 
253  binval = SPI_getbinval(*tuple, *tupdesc, info.colNumber, &isNull);
254  if (!(info.type == BPCHAROID)) {

263  value = default_value;

566 TTimestamp
567 timestamp_without_timezone(TTimestamp timestamp) {
568  /*
569  * step 1
570  */
571  Timestamp date;
572  Timestamp time = timestamp;
573  TMODULO(time, date, USECS_PER_DAY);
574  if (time < INT64CONST(0)) {
575  time += USECS_PER_DAY;
576  date -= 1;
577  }
578  date += POSTGRES_EPOCH_JDATE;
579  /* Julian day routine does not work for negative Julian days */
580  if (date < 0 || date > (Timestamp) INT_MAX) {
581  ereport(ERROR,
582  (errcode(ERRCODE_INTERNAL_ERROR),
583  errmsg("Julian day routine does not work for negative Julian days")));
584  }
585 
586  /*
587  * using structure from time.h to store values
588  */
589  struct tm info;
590  fsec_t fsec;
591 
592  /*
593  * calling postgres functions
594  */
595  j2date((int) date, &info.tm_year, &info.tm_mon, &info.tm_mday);
596  dt2time(time, &info.tm_hour, &info.tm_min, &info.tm_sec, &fsec);
597 
598  /*
599  * adjust values before calling mktime
600  */
601  info.tm_isdst = -1;
602  info.tm_year = info.tm_year - 1900;
603  info.tm_mon = info.tm_mon - 1;
604 
605  /*
606  * mktime & timezone are defined in time.h
607  */
608  return mktime(&info) - timezone;
609 }

625 void pgr_fetch_column_info(
626  Column_info_t info[],
627  int info_size) {
628  for (int i = 0; i < info_size; ++i) {
629  if (fetch_column_info(&info[i])) {
630  switch (info[i].eType) {
631  case ANY_INTEGER:
632  check_any_integer_type(info[i]);
633  break;
634  case ANY_NUMERICAL:
635  check_any_numerical_type(info[i]);
636  break;
637  case TEXT:
638  check_text_type(info[i]);
639  break;
640  case CHAR1:
641  check_char_type(info[i]);
642  break;
643  case ANY_INTEGER_ARRAY:
644  check_any_integerarray_type(info[i]);
645  break;
646  case TIMESTAMP:
647  check_timestamp_type(info[i]);
648  break;
649  case INTERVAL:
650  check_interval_type(info[i]);
651  break;
652  default:
653  elog(ERROR, "Unknown type of column %s", info[i].name);
654  }
655  }
656  }

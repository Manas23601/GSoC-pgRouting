83 
84  Vroom_rt **return_tuples,
85  size_t *return_count,
86 
87  char ** log_msg,
88  char ** notice_msg,
89  char ** err_msg) {
90  std::ostringstream log;
91  std::ostringstream err;
92  std::ostringstream notice;
93  try {
94  pgassert(!(*log_msg));
95  pgassert(!(*notice_msg));
96  pgassert(!(*err_msg));
97  pgassert(!(*return_tuples));
98  pgassert(!(*return_count));
99  pgassert(jobs || shipments);
100  pgassert(vehicles);
101  pgassert(matrix_cells_arr);
102  pgassert(total_jobs || total_shipments);
103  pgassert(total_vehicles);
104  pgassert(total_cells);
105 
106  Identifiers<Id> location_ids;
107 
108  for (size_t i = 0; i < total_jobs; ++i) {
109  location_ids += jobs[i].location_index;
110  }
111 
112  for (size_t i = 0; i < total_shipments; ++i) {
113  location_ids += shipments[i].p_location_index;
114  location_ids += shipments[i].d_location_index;
115  }
116 
117  for (size_t i = 0; i < total_vehicles; ++i) {
118  if (vehicles[i].start_index != -1) {
119  location_ids += vehicles[i].start_index;
120  }
121  if (vehicles[i].end_index != -1) {
122  location_ids += vehicles[i].end_index;
123  }
124  }
125 
126  vrprouting::base::Base_Matrix time_matrix(matrix_cells_arr, total_cells, location_ids);
127 
128  /*
129  * Verify matrix cells preconditions
130  */
131  if (!time_matrix.has_no_infinity()) {
132  (*return_tuples) = NULL;
133  (*return_count) = 0;
134  err << "An Infinity value was found on the Matrix. Might be missing information of a node";
135  *err_msg = pgr_msg(err.str());
136  return;
137  }
138 
139  /*
140  * Verify size of matrix cell lies in the limit
141  */
142  if (time_matrix.size() > (std::numeric_limits<vroom::Index>::max)()) {
143  (*return_tuples) = NULL;
144  (*return_count) = 0;
145  err << "The size of time matrix exceeds the limit";
146  *err_msg = pgr_msg(err.str());
147  return;
148  }
149 
150  vrprouting::Vrp_vroom_problem problem;
151  problem.add_matrix(time_matrix);
152  problem.add_vehicles(vehicles, total_vehicles,
153  breaks, total_breaks,
154  breaks_tws, total_breaks_tws);
155  problem.add_jobs(jobs, total_jobs,
156  jobs_tws, total_jobs_tws);
157  problem.add_shipments(shipments, total_shipments,
158  shipments_tws, total_shipments_tws);
159 
160  std::vector < Vroom_rt > results = problem.solve();
161 
162  auto count = results.size();
163  if (count == 0) {
164  (*return_tuples) = NULL;
165  (*return_count) = 0;
166  notice << "No results found";
167  *notice_msg = notice.str().empty()?
168  *notice_msg :
169  pgr_msg(notice.str().c_str());
170  return;
171  }
172 
173  (*return_tuples) = pgr_alloc(count, (*return_tuples));
174  for (size_t i = 0; i < count; i++) {
175  *((*return_tuples) + i) = results[i];
176  }
177 
178  (*return_count) = count;
179 
180  pgassert(*err_msg == NULL);
181  *log_msg = log.str().empty()?
182  *log_msg :
183  pgr_msg(log.str().c_str());
184  *notice_msg = notice.str().empty()?
185  *notice_msg :
186  pgr_msg(notice.str().c_str());
187  } catch (AssertFailedException &except) {
188  (*return_tuples) = pgr_free(*return_tuples);
189  (*return_count) = 0;
190  err << except.what();
191  *err_msg = pgr_msg(err.str().c_str());
192  *log_msg = pgr_msg(log.str().c_str());
193  } catch (const vroom::Exception &except) {
194  (*return_tuples) = pgr_free(*return_tuples);
195  (*return_count) = 0;
196  err << except.message;
197  *err_msg = pgr_msg(err.str().c_str());
198  *log_msg = pgr_msg(log.str().c_str());
199  } catch (std::exception &except) {
200  (*return_tuples) = pgr_free(*return_tuples);
201  (*return_count) = 0;
202  err << except.what();
203  *err_msg = pgr_msg(err.str().c_str());
204  *log_msg = pgr_msg(log.str().c_str());
205  } catch(...) {
206  (*return_tuples) = pgr_free(*return_tuples);
207  (*return_count) = 0;
208  err << "Caught unknown exception!";
209  *err_msg = pgr_msg(err.str().c_str());
210  *log_msg = pgr_msg(log.str().c_str());
211  }
212 }

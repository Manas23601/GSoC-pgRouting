50 get_distance(std::pair<Coordinate, Coordinate> p1 , std::pair<Coordinate, Coordinate> p2) {
51  auto dx = p1.first - p2.first;
52  auto dy = p1.second - p2.second;
53  return std::sqrt(dx * dx + dy * dy);
54 }

159 Idx
160 Base_Matrix::get_index(Id id) const {
161  /*
162  * binary search the stored identifiers
163  */
164  auto pos = std::lower_bound(m_ids.begin(), m_ids.end(), id);
165  if (pos == m_ids.end()) {
166  std::ostringstream msg;

168  pgassertwm(false, msg.str());

173  /*
174  * return the index found
175  */
176  return static_cast<Idx>(pos - m_ids.begin());

193 Base_Matrix::Base_Matrix(
194  Matrix_cell_t *data_costs, size_t size_matrix,
195  const Identifiers<Id>& node_ids,
196  Multiplier multiplier) {
197  /*
198  * Sets the selected nodes identifiers
199  */
200  m_ids.insert(m_ids.begin(), node_ids.begin(), node_ids.end());
201 
202  /*
203  * Create matrix
204  */
205  m_time_matrix.resize(
206  m_ids.size(),
207  std::vector<TInterval>(
208  m_ids.size(),
209  /*
210  * Set initial values to infinity
211  */
212  (std::numeric_limits<TInterval>::max)()));
213 
214  Identifiers<Idx> inserted;
215  /*
216  * Cycle the matrix data
217  */
218  for (size_t i = 0; i < size_matrix; ++i) {
219  auto data = data_costs[i];
220  /*
221  * skip if row is not from selected nodes
222  */
223  if (!(has_id(data.from_vid) && has_id(data.to_vid))) continue;
224 
225  /*
226  * Save the information
227  */
228  m_time_matrix[get_index(data.from_vid)][get_index(data.to_vid)] =
229  static_cast<TInterval>(static_cast<Multiplier>(data.cost) * multiplier);
230 
231  /*
232  * If the opposite direction is infinity insert the same cost
233  */
234  if (m_time_matrix[get_index(data.to_vid)][get_index(data.from_vid)] == (std::numeric_limits<TInterval>::max)()) {
235  m_time_matrix[get_index(data.to_vid)][get_index(data.from_vid)] =
236  m_time_matrix[get_index(data.from_vid)][get_index(data.to_vid)];
237  }
238  }
239 
240  /*
241  * Set the diagonal values to 0
242  */
243  for (size_t i = 0; i < m_time_matrix.size(); ++i) {
244  m_time_matrix[i][i] = 0;
245  }
246 }

251 Base_Matrix::Base_Matrix(const std::map<std::pair<Coordinate, Coordinate>, Id> &euclidean_data, Multiplier multiplier) {
252  m_ids.reserve(euclidean_data.size());
253  for (const auto &e : euclidean_data) {
254  m_ids.push_back(e.second);
255  }
256  m_time_matrix.resize(
257  m_ids.size(),
258  std::vector<TInterval>(
259  m_ids.size(),
260  (std::numeric_limits<TInterval>::max)()));
261 
262  for (const auto &from : euclidean_data) {
263  for (const auto &to : euclidean_data) {
264  auto from_id = get_index(from.second);
265  auto to_id = get_index(to.second);
266  m_time_matrix[from_id][to_id] =
267  static_cast<TInterval>(static_cast<Multiplier>(detail::get_distance(from.first, to.first)) * multiplier);
268  m_time_matrix[to_id][from_id] = m_time_matrix[from_id][to_id];
269  }
270  }
271 
272  for (size_t i = 0; i < m_time_matrix.size(); ++i) {
273  m_time_matrix[i][i] = 0;
274  }
275 }

354 size_t
355 Base_Matrix::fix_triangle_inequality(size_t depth) {
356  if (depth > m_time_matrix.size()) return depth;
357  for (auto & i : m_time_matrix) {
358  for (size_t j = 0; j < m_time_matrix.size(); ++j) {
359  for (size_t k = 0; k < m_time_matrix.size(); ++k) {
360  if (i[k] > (i[j] + m_time_matrix[j][k])) {
361  i[k] = i[j] + m_time_matrix[j][k];
362  return fix_triangle_inequality(++depth);
363  }
364  }
365  }
366  }
367  return depth;
368 }

373 bool
374 Fleet::is_fleet_ok() const {
375  ENTERING(msg());
376  if (!msg().get_error().empty()) return false;
377  for (auto truck : m_trucks) {
378  if (!truck.is_ok()) {
379  msg().error << "Illegal values found on vehicle";
380  msg().log << "On vehicle " << truck.id()
381  << " a condition is not met, verify that:\n"
382  << "- start_open <= start_close\n"
383  << "- end_open <= end_close\n"
384  << "- capacity > 0\n";
385  return false;
386  }
387 
388  if (!(truck.start_site().is_start()
389  && truck.end_site().is_end())) {
390  pgassertwm(false, "should never pass through here");
391  msg().error << "Illegal values found on vehicle";
392  return false;
393  }
394  if (!truck.is_feasable()) {
395  msg().error << "Truck is not feasible";
396  return false;
397  }
398  }
399  EXITING(msg());
400  return true;
401 }

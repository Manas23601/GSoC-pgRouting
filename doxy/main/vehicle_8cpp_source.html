64 size_t
65 Vehicle::insert(std::pair<POS, POS> position_limits, const Vehicle_node &node) {
66  invariant();
67  pgassert(position_limits.first <= m_path.size());
68  pgassert(position_limits.second <= m_path.size());
69 
70  auto low = position_limits.first;
71  auto high = position_limits.second;
72  auto best = low;
73 
74 
75  insert(low, node);
76 
77  Vehicle::Cost best_cost(cost());
78 
79 
80  while (low < high) {
81  swap(low, low + 1);
82  ++low;
83  if (cost_compare(best_cost, cost())) {
84  best_cost = cost();
85  best = low;
86  }
87  }
88  return best;
89 
90  pgassert(best < m_path.size());
91  pgassert(m_path[best].idx() == node.idx());
92  invariant();
93 }

98 bool
99 Vehicle::cost_compare(const Cost &lhs, const Cost &rhs) const {
100  /*
101  * capacity violations
102  */
103  if (std::get<1>(lhs) < std::get<1>(rhs))
104  return true;
105  if (std::get<1>(lhs) > std::get<1>(rhs))
106  return false;
107 
108  /*
109  * time window violations
110  */
111  if (std::get<0>(lhs) < std::get<0>(rhs))
112  return true;
113  if (std::get<0>(lhs) > std::get<0>(rhs))
114  return false;
115 
116  /*
117  * waiting time
118  */
119  if (std::get<3>(lhs) < std::get<3>(rhs))
120  return true;
121  if (std::get<3>(lhs) > std::get<3>(rhs))
122  return false;
123 
124  /*
125  * duration
126  */
127  if (std::get<4>(lhs) < std::get<4>(rhs))
128  return true;
129  if (std::get<4>(lhs) > std::get<4>(rhs))
130  return false;
131 
132  /*
133  * truck size
134  */
135  if (std::get<2>(lhs) < std::get<2>(rhs))
136  return true;
137  if (std::get<2>(lhs) > std::get<2>(rhs))
138  return false;
139 
140  return false;
141 }

198 double
199 Vehicle::deltaTime(const Vehicle_node &node, POS pos) const {
200  /*
201  * .... POS POS+1 ....
202  * .... POS node POS+1 ....
203  *
204  */
205  auto prev = m_path[pos-1];
206  auto next = m_path[pos];
207  auto original_time = next.travel_time();
208  auto tt_p_n = prev.travel_time_to(node, speed());
209  tt_p_n = node.is_early_arrival(prev.departure_time() + tt_p_n) ?
210  node.closes() - prev.departure_time()
211  : tt_p_n;
212 
213  auto tt_n_x = node.travel_time_to(next, speed());
214  tt_p_n = next.is_early_arrival(
215  prev.departure_time() + tt_p_n + node.service_time() + tt_n_x) ?
216  next.closes() - (prev.departure_time() + tt_p_n + node.service_time())
217  : tt_n_x;
218 
219  return (tt_p_n + tt_n_x) - original_time;
220 }

47 Vehicle_pickDeliver::get_worse_order(
48  Identifiers<size_t> orders) const {
49  invariant();
50  pgassert(!empty());
51 
52  // auto orders(of_this_subset);
53  auto worse_order(m_orders[*orders.begin()]);
54  auto delta_duration((std::numeric_limits<double>::max)());
55  auto curr_duration(duration());
56  while (!orders.empty()) {
57  auto truck(*this);
58  auto order = m_orders[*orders.begin()];
59  pgassert(truck.has_order(order));
60  orders -= order.idx();
61  truck.erase(order);
62  auto delta = truck.duration() - curr_duration;
63  if (delta < delta_duration) {
64  worse_order = order;
65  delta_duration = delta;
66  }
67  }
68  return worse_order;
69 }

110 bool
111 Vehicle_pickDeliver::insert(const Order &order) {
112  invariant();
113  pgassert(!has_order(order));
114 
115  auto pick_pos(position_limits(order.pickup()));
116  auto deliver_pos(position_limits(order.delivery()));
117 #ifndef NDEBUG
118  std::ostringstream err_log;
119  err_log << "\n\tpickup limits (low, high) = ("
120  << pick_pos.first << ", "
121  << pick_pos.second << ") "
122  << "\n\tdeliver limits (low, high) = ("
123  << deliver_pos.first << ", "
124  << deliver_pos.second << ") "
125  << "\noriginal" << tau();
126 #endif
127 
128  if (pick_pos.second < pick_pos.first) {
129  /*
130  * pickup generates twv evrywhere
131  */
132  return false;
133  }
134 
135  if (deliver_pos.second < deliver_pos.first) {
136  /*
137  * delivery generates twv evrywhere
138  */
139  return false;
140  }
141  /*
142  * Because delivery positions were estimated without
143  * the pickup:
144  * - increase the upper limit position estimation
145  */
146  ++deliver_pos.first;
147  ++deliver_pos.second;
148 
149 
150  auto d_pos_backup(deliver_pos);
151  auto best_pick_pos = m_path.size();
152  auto best_deliver_pos = m_path.size() + 1;
153  auto current_duration(duration());
154  auto min_delta_duration = (std::numeric_limits<double>::max)();
155  auto found(false);
156  pgassertwm(!has_order(order), err_log.str());
157  while (pick_pos.first <= pick_pos.second) {
158 #ifndef NDEBUG
159  err_log << "\n\tpickup cycle limits (low, high) = ("
160  << pick_pos.first << ", "
161  << pick_pos.second << ") ";
162 #endif
163  Vehicle::insert(pick_pos.first, order.pickup());
164 #ifndef NDEBUG
165  err_log << "\npickup inserted: " << tau();
166 #endif
167 
168  if (deliver_pos.first <= pick_pos.first) deliver_pos.first = pick_pos.first + 1;
169 
170  while (deliver_pos.first <= deliver_pos.second) {
171  Vehicle::insert(deliver_pos.first, order.delivery());
172  m_orders_in_vehicle += order.idx();
173  pgassertwm(has_order(order), err_log.str());
174 #ifndef NDEBUG
175  err_log << "\ndelivery inserted: " << tau();
176 #endif
177  if (is_feasable()) {
178  pgassert(is_feasable());
179  auto delta_duration = duration() - current_duration;
180  if (delta_duration < min_delta_duration) {
181 #ifndef NDEBUG
182  err_log << "\nsuccess" << tau();
183 #endif
184  min_delta_duration = delta_duration;
185  best_pick_pos = pick_pos.first;
186  best_deliver_pos = deliver_pos.first;
187  found = true;
188  }
189  }
190  Vehicle::erase(deliver_pos.first);
191 #ifndef NDEBUG
192  err_log << "\ndelivery erased: " << tau();
193 #endif
194  ++deliver_pos.first;
195  }
196  Vehicle::erase(pick_pos.first);
197 #ifndef NDEBUG
198  err_log << "\npickup erased: " << tau();
199 #endif
200  m_orders_in_vehicle -= order.idx();
201  pgassertwm(!has_order(order), err_log.str());
202 
203  deliver_pos = d_pos_backup;
204 #ifndef NDEBUG
205  err_log << "\n\trestoring deliver limits (low, high) = ("
206  << deliver_pos.first << ", "
207  << deliver_pos.second << ") ";
208 #endif
209  ++pick_pos.first;
210  }
211  pgassertwm(!has_order(order), err_log.str());
212  if (!found) {
213  /* order causes twv or cv */
214  return false;
215  }
216  Vehicle::insert(best_pick_pos, order.pickup());
217  Vehicle::insert(best_deliver_pos, order.delivery());
218 
219  m_orders_in_vehicle += order.idx();
220  pgassertwm(is_feasable(), err_log.str());
221  pgassertwm(has_order(order), err_log.str());
222  pgassertwm(!has_cv(), err_log.str());
223  invariant();
224  return true;
225 }

264 void
265 Vehicle_pickDeliver::do_while_feasable(
266  Initials_code kind,
267  Identifiers<size_t> &unassigned,
268  Identifiers<size_t> &assigned) {
269  pgassert(is_feasable());
270 #if 0
271  msg.log << "unasigned" << unassigned << "\n";
272  msg.log << "m_feasable_orders" << m_feasable_orders << "\n";
273 #endif
274  auto current_feasable = m_feasable_orders * unassigned;
275 
276  while (!current_feasable.empty()) {
277 #if 0
278  msg.log << "current_feasable" << current_feasable << "\n";
279 #endif
280  auto order = m_orders[current_feasable.front()];
281 
282  switch (kind) {
283  case OnePerTruck:
284  push_back(order);
285  pgassert(is_feasable());
286  assigned += order.idx();
287  unassigned -= order.idx();
288  invariant();
289  return;
290  break;
291  case FrontTruck:
292  push_front(order);
293  break;
294  case BackTruck:
295  push_back(order);
296  break;
297  case BestInsert:
298  insert(order);
299  break;
300  case BestBack:
301  order = m_orders[m_orders.find_best_J(current_feasable)];
302  insert(order);
303  break;
304  case BestFront:
305  order = m_orders[m_orders.find_best_I(current_feasable)];
306  insert(order);
307  break;
308  case OneDepot:
309  semiLIFO(order);
310  break;
311  default: pgassert(false);
312  }
313 
314  if (orders_size() == 1 && !is_feasable()) {
315  pgassert(false);
316  }
317 
318  if (!is_feasable()) {
319  erase(order);
320  } else if (has_order(order)) {
321  assigned += order.idx();
322  unassigned -= order.idx();
323  if (kind == BestBack) {
324  current_feasable = m_orders[order.idx()].subsetJ(
325  current_feasable);
326  }
327  if (kind == BestFront) {
328  current_feasable = m_orders[order.idx()].subsetI(
329  current_feasable);
330  }
331  }
332 
333  current_feasable -= order.idx();
334  invariant();
335  }
336 
337  pgassert(is_feasable());
338  invariant();
339 }

427 bool
428 Vehicle_pickDeliver::semiLIFO(const Order &order) {
429  invariant();
430  pgassert(!has_order(order));
431 
432  /*
433  * Insert pick up as first picked
434  */
435  Vehicle::insert(1, order.pickup());
436 
437  auto deliver_pos(drop_position_limits(order.delivery()));
438 
439  /*
440  * delivery generates twv in all positions
441  */
442  if (deliver_pos.second < deliver_pos.first) {
443  /*
444  * Remove inserted pickup
445  */
446  Vehicle::erase(1);
447  invariant();
448  return false;
449  }
450 
451  pgassert(!has_order(order));
452 
453  while (deliver_pos.first <= deliver_pos.second) {
454  Vehicle::insert(deliver_pos.second, order.delivery());
455 
456  if (is_feasable() && !m_path[deliver_pos.second + 1].is_pickup()) {
457  /*
458  * Found a position to insert the delivery
459  */
460 
461 
462  m_orders_in_vehicle += order.idx();
463 
464  /*
465  * There is one more order in the vehicle
466  */
467  pgassert(has_order(order));
468  pgassert(is_feasable());
469  pgassert(!has_cv());
470  pgassert(!has_twv());
471  pgassert(has_order(order));
472  invariant();
473  return true;
474  }
475 
476  /*
477  * This position in path is not suitable
478  */
479  Vehicle::erase(deliver_pos.second);
480 
481  /*
482  * got to next position
483  */
484  --deliver_pos.second;
485  }
486 
487  /*
488  * Order could not be inserted
489  */
490  Vehicle::erase(1);
491 
492  pgassert(!has_order(order));
493  invariant();
494  return false;
495 }

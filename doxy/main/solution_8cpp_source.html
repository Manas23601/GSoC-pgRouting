196 bool
197 Solution::operator<(const Solution &s_rhs) const {
198  Vehicle::Cost lhs(cost());
199  Vehicle::Cost rhs(s_rhs.cost());
200 
201  /*
202  * capacity violations
203  */
204  if (std::get<0>(lhs) < std::get<0>(rhs))
205  return true;
206  if (std::get<0>(lhs) > std::get<0>(rhs))
207  return false;
208 
209  /*
210  * time window violations
211  */
212  if (std::get<1>(lhs) < std::get<1>(rhs))
213  return true;
214  if (std::get<1>(lhs) > std::get<1>(rhs))
215  return false;
216 
217  /*
218  * fleet size
219  */
220  if (std::get<2>(lhs) < std::get<2>(rhs))
221  return true;
222  if (std::get<2>(lhs) > std::get<2>(rhs))
223  return false;
224 
225  /*
226  * waiting time
227  */
228  if (std::get<3>(lhs) < std::get<3>(rhs))
229  return true;
230  if (std::get<3>(lhs) > std::get<3>(rhs))
231  return false;
232 
233  /*
234  * duration
235  */
236  if (std::get<4>(lhs) < std::get<4>(rhs))
237  return true;
238  if (std::get<4>(lhs) > std::get<4>(rhs))
239  return false;
240 
241  return false;
242 }

156 bool
157 PD_Orders::is_valid(double speed) const {
158  for (const auto &o : m_orders) {
159  if (!o.is_valid(speed)) {
160  return false;
161  }
162  pgassert(o.pickup().is_pickup());
163  pgassert(o.delivery().is_delivery());
164  /* P -> D */
165  pgassert(o.delivery().is_compatible_IJ(o.pickup(), speed));
166  }
167  return true;

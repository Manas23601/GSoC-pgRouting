135 /*
136  * .. to ... from ....
137  */
138 bool
139 Optimize::swap_worse(Vehicle_pickDeliver &to, Vehicle_pickDeliver &from) {
140  /*
141  * working on a copy of the vehicles
142  */
143  auto from_truck = from;
144  auto to_truck = to;
145 
146  auto swapped = false;
147 
148  /*
149  * To avoid invalidation of cycles
150  */
151  auto o_from_orders = from_truck.orders_in_vehicle();
152  auto o_to_orders = to_truck.orders_in_vehicle();
153 
154  pgassert((o_from_orders * o_to_orders).empty());
155 
156  for (auto from_orders = from_truck.orders_in_vehicle();
157  !from_orders.empty();
158  from_orders.pop_front()) {
159  auto from_order = from_truck.orders()[from_orders.front()];
160 
161  pgassert(from_truck.has_order(from_order));
162 
163  if (move_order(from_order, from_truck, to_truck)) {
164  pgassert(!from_truck.has_order(from_order));
165  pgassert(to_truck.has_order(from_order));
166  /*
167  * The order could be moved to "to" truck
168  */
169  to = to_truck;
170  from = from_truck;
171  /*
172  * go to next order
173  */
174  pgassert(!from.has_order(from_order));
175  pgassert(to.has_order(from_order));
176  pgassert(!from_truck.has_order(from_order));
177  pgassert(to_truck.has_order(from_order));
178  pgassert(!(from.has_order(from_order) && to.has_order(from_order)));
179  pgassert(from.has_order(from_order) || to.has_order(from_order));
180  continue;
181  }
182 
183  pgassert(from_truck.has_order(from_order));
184  auto curr_from_duration = from_truck.duration();
185 
186  for (auto to_order_id : o_to_orders) {
187  auto to_order = to.orders()[to_order_id];
188  /*
189  * The orders might have being swapped before
190  */
191  if (!to_truck.has_order(to_order)) continue;
192  // if (!from_truck.has_order(from_order)) break; // TODO delete line
193 
194  pgassert(from_truck.has_order(from_order));
195  pgassert(to_truck.has_order(to_order));
196 
197  auto curr_to_duration = to_truck.duration();
198 
199  /*
200  * delete from_order, and to order from their trucks
201  */
202  pgassert(from_truck.has_order(from_order));
203  from_truck.erase(from_order);
204  pgassert(to_truck.has_order(to_order));
205  to_truck.erase(to_order);
206 
207  /*
208  * insert them in the other truck
209  */
210  if (this->get_kind() == OneDepot) {
211  from_truck.semiLIFO(to_order);
212  to_truck.semiLIFO(from_order);
213  } else {
214  from_truck.insert(to_order);
215  to_truck.insert(from_order);
216  }
217 
218  pgassert((from_truck.has_order(to_order) && from_truck.is_feasable()) || !from_truck.has_order(to_order));
219  pgassert((to_truck.has_order(from_order) && to_truck.is_feasable()) || !to_truck.has_order(from_order));
220 
221  if (from_truck.has_order(to_order) && to_truck.has_order(from_order)) {
222  auto new_from_duration = from_truck.duration();
223  auto new_to_duration = to_truck.duration();
224 
225  auto estimated_delta =
226  - (curr_from_duration + curr_to_duration)
227  + (new_to_duration + new_from_duration);
228 
229  auto estimated_duration = duration() + estimated_delta;
230 
231  /*
232  * Can swap when:
233  * - or from_truck duration is reduced
234  * - the total fleet duration is reduced
235  * - or the new fleet duration is better than best_solution duration
236  */
237  if (new_from_duration < curr_from_duration ||
238  estimated_delta < 0 ||
239  estimated_duration < best_solution.duration()) {
240  /*
241  * this acctually makes the swapping
242  */
243  to = to_truck;
244  from = from_truck;
245 
246  pgassert(!(from.has_order(from_order) && to.has_order(from_order)));
247  pgassert(from.has_order(from_order) || to.has_order(from_order));
248  pgassert(!(from.has_order(to_order) && to.has_order(to_order)));
249  pgassert(from.has_order(to_order) || to.has_order(to_order));
250 
251  swapped = true;
252  break;
253  }
254  }
255  /*
256  * Can't swap, restore vehicles
257  */
258  to_truck = to;
259  from_truck = from;
260 
261  pgassert(!(from.has_order(from_order) && to.has_order(from_order)));
262  pgassert(from.has_order(from_order) || to.has_order(from_order));
263  pgassert(!(from.has_order(to_order) && to.has_order(to_order)));
264  pgassert(from.has_order(to_order) || to.has_order(to_order));
265  }
266  }
267 
268  return false && swapped;
269 }

322 /*
323  * from_truck: position of the truck where the order is
324  * to truck: truck to put the order
325  */
326 bool
327 Optimize::swap_order(
328  const Order from_order,
329  Vehicle_pickDeliver &from_truck,
330  const Order to_order,
331  Vehicle_pickDeliver &to_truck) {
332  if (!from_truck.has_order(from_order)
333  || !to_truck.has_order(to_order)) {
334  return false;
335  }
336 
337  pgassert(from_truck.has_order(from_order));
338  pgassert(to_truck.has_order(to_order));
339 
340  from_truck.erase(from_order);
341  to_truck.erase(to_order);
342 
343  /*
344  * insert them in the other truck
345  */
346  if (this->get_kind() == OneDepot) {
347  from_truck.semiLIFO(to_order);
348  to_truck.semiLIFO(from_order);
349  } else {
350  from_truck.insert(to_order);
351  to_truck.insert(from_order);
352  }
353 
354  pgassert((from_truck.has_order(to_order) && from_truck.is_feasable()) || !from_truck.has_order(to_order));
355  pgassert((to_truck.has_order(from_order) && to_truck.is_feasable()) || !to_truck.has_order(from_order));
356 
357  if (!from_truck.has_order(to_order) || !to_truck.has_order(from_order)) {
358  /*
359  * swap generates a violation: restore trucks
360  */
361  if (from_truck.has_order(to_order)) from_truck.erase(to_order);
362  if (to_truck.has_order(from_order)) to_truck.erase(from_order);
363 
364  pgassert(!from_truck.has_order(to_order));
365  pgassert(!from_truck.has_order(from_order));
366  pgassert(!to_truck.has_order(from_order));
367  pgassert(!to_truck.has_order(to_order));
368 
369  /*
370  * insert them again in the truck
371  */
372  if (this->get_kind() == OneDepot) {
373  from_truck.semiLIFO(from_order);
374  to_truck.semiLIFO(to_order);
375  } else {
376  from_truck.insert(from_order);
377  to_truck.insert(to_order);
378  }
379  pgassert((from_truck.has_order(from_order) && from_truck.is_feasable()) || !from_truck.has_order(from_order));
380  pgassert((to_truck.has_order(to_order) && to_truck.is_feasable()) || !to_truck.has_order(to_order));
381 
382  return false;
383  }
384 
385  pgassert(from_truck.has_order(to_order));
386  pgassert(to_truck.has_order(from_order));
387  return true;
388 }

509 /*
510  * from_truck trying to make from_truck's duration smaller
511  * - maybe all orders can be moved
512  * - if that is the case, the from_truck could be removed
513  *
514  * Deleting an order on the from_truck
515  * - number of truck remains the same
516  * - from_truk duration() can not get larger
517  * - the overall duration can get larger
518  *
519  */
520 bool
521 Optimize::move_reduce_cost(
522  Vehicle_pickDeliver &from,
523  Vehicle_pickDeliver &to) {
524  pgassert(&from != &to);
525 
526  auto from_truck = from;
527  auto to_truck = to;
528  /*
529  * don't move to empty truck
530  */
531  if (to_truck.empty()) return false;
532 
533  /*
534  * don't move from a real truck to a phoney truck
535  */
536  if (!from_truck.is_phony() && to_truck.is_phony()) {
537  return false;
538  }
539 
540  auto moved = false;
541 
542  auto from_orders = from_truck.orders_in_vehicle();
543  for (const auto o_id : from_orders) {
544  /*
545  * removing an order decreases the duration
546  */
547  auto order = from_truck.orders()[o_id];
548 
549  auto curr_duration = from_truck.duration() + to_truck.duration();
550  /*
551  * insert it in the "to" truck
552  */
553  pgassert(!to_truck.has_order(order));
554  this->get_kind() == OneDepot?
555  to_truck.semiLIFO(order) :
556  to_truck.insert(order);
557  pgassert((to_truck.has_order(order) && to_truck.is_feasable()) || !to_truck.has_order(order));
558 
559  if (to_truck.has_order(order)) {
560  from_truck.erase(order);
561 
562  auto new_duration = from_truck.duration() + to_truck.duration();
563 
564  /*
565  * cost is reduced
566  */
567  if (new_duration < curr_duration
568  || from_truck.empty()
569  || new_duration < best_solution.duration()) {
570  moved = true;
571  save_if_best();
572  continue;
573  }
574 
575  /*
576  * cost is not reduced
577  * revert changes
578  */
579  to_truck.erase(order);
580  this->get_kind() == OneDepot?
581  from_truck.semiLIFO(order) :
582  from_truck.insert(order);
583  }
584  }
585  return moved;
586 }

663  * Optimize decreasing truck
664  *
665  * - Objective: try to remove truck with less duration
666  * - Secundary objective, acts like a shake operation
667  *
668  */
669 void
670 Optimize::decrease_truck() {
671  bool decreased(false);
672  for (size_t i = 1; i < fleet.size(); ++i) {
673  decreased = decrease_truck(i) || decreased;
674  }
675  if (decreased) {
676  delete_empty_truck();
677  save_if_best();
678  decrease_truck();
679  }
680  save_if_best();
681 }

683 bool
684 Optimize::decrease_truck(size_t cycle) {
685  auto position = cycle;
686  for (auto orders = fleet[position].orders_in_vehicle();
687  !orders.empty();
688  orders.pop_front()) {
689  /* Step 2: grab an order */
690  auto order = fleet[position].orders()[orders.front()];
691  pgassert(order.idx() == orders.front());
692 
693 
694  /* Step 3:
695  * cycle the fleet
696  * insert in first truck possible
697  */
698 
699  for (size_t i = 0; i < position; ++i) {
700  fleet[i].insert(order);
701  pgassert((fleet[i].has_order(order) && fleet[i].is_feasable()) || !fleet[i].has_order(order));
702  if (fleet[i].has_order(order)) {
703  /*
704  * delete the order from the current truck
705  */
706  fleet[position].erase(order);
707  break;
708  }
709  }
710  }
711  return fleet[position].orders_in_vehicle().empty();
712 }
